import java.io.IOException;
import java.util.Arrays;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet({"/BulkStudentActionServlet", "/Admin/BulkStudentActionServlet"})
public class BulkStudentActionServlet extends HttpServlet {

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String selectAllStudents = request.getParameter("selectAllStudents");
        String[] studentIds = request.getParameterValues("studentIds");

        request.setAttribute("selectAllStudents", selectAllStudents);
        request.setAttribute("studentIds", studentIds);

        System.out.println("Selected students: " + Arrays.toString(studentIds));
        response.sendRedirect(request.getContextPath() + "/Admin/Students.html");
    }
}
