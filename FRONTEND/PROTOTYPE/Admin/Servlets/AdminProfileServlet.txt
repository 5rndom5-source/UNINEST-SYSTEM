import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet({"/AdminProfileServlet", "/Admin/AdminProfileServlet"})
public class AdminProfileServlet extends HttpServlet {

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String profileAction = request.getParameter("profileAction");

        if ("settings".equals(profileAction)) {
            response.sendRedirect(request.getContextPath() + "/Admin/Settings.html");
        } else if ("logout".equals(profileAction)) {
            response.sendRedirect(request.getContextPath() + "/Admin/Logout.html");
        } else {
            response.sendRedirect(request.getContextPath() + "/Admin/Dashboard.html");
        }
    }
}
