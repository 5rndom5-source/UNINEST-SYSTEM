import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet({"/StudentServlet", "/Admin/StudentServlet"})
public class StudentServlet extends HttpServlet {

    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        String studentSearch = request.getParameter("studentSearch");
        String studentSearchBtn = request.getParameter("studentSearchBtn");
        String studentStatus = request.getParameter("studentStatus");
        String program = request.getParameter("program");
        String page = request.getParameter("page");

        request.setAttribute("studentSearch", studentSearch);
        request.setAttribute("studentSearchBtn", studentSearchBtn);
        request.setAttribute("studentStatus", studentStatus);
        request.setAttribute("program", program);
        request.setAttribute("page", page);

        // Add SELECT students WHERE filters logic here.
        response.sendRedirect(request.getContextPath() + "/Admin/Students.html");
    }
}
